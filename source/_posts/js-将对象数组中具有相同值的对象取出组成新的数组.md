/**
* arr 要转化的数组
* key 根据某一键转化
*/
function sortArr(arr, key) {
  let newArr = [],
      tempArr = [],
      temp;

  // 按照特定的参数将数组排序，将具有相同值的排在一起
  arr = arr.sort(function(a, b) {
      let s = a[key],
          t = b[key];

      return s < t ? -1 : 1;
  });
  console.log(arr,"排序之后的数组");

  if ( arr.length ){
      temp = arr[0][key];
  }

  // 将相同类别的对象添加到统一个数组
  for (let i in arr) {
      if ( arr[i][key] === temp ){
          tempArr.push( arr[i] );
      } else {
          temp = arr[i][key];
          newArr.push(tempArr);
          tempArr = [arr[i]];
      }
  }
  // 将最后的内容推出新数组
  newArr.push(tempArr);
  return newArr;
}
